id -g squid => to get Gid Ans (23)  
iptables -t nat -A OUTPUT -p tcp --dport 80 -m owner ! --gid-owner 23 -j REDIRECT --to-port 3129
iptables -t nat -A OUTPUT -p tcp --dport 443 -m owner ! --gid-owner 23 -j REDIRECT --to-port 3129
