#!/bin/bash

echo 'Content-type: text/html'
echo ''
echo 'welcome'

chmod u+r+w+x,g+r+x,o+r+x trail.cgi
ls -lZ
systemctl start httpd
http://localhost/cgi-bin/trail.cgi
